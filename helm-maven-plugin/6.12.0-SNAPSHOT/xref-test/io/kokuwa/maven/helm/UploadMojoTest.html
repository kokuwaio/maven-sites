344 	private void assertUpload(UploadMojo mojo, RequestMethod method, String path, String authorization) {
345 
346 		RequestPatternBuilder requestPattern;
347 		if (authorization == null) {
348 			mock.stubFor(WireMock.any(WireMock.anyUrl()).willReturn(WireMock.ok()));
349 			requestPattern = RequestPatternBuilder.allRequests();
350 		} else {
351 			mock.stubFor(WireMock.any(WireMock.anyUrl())
352 					.withHeader(HttpHeaders.AUTHORIZATION, WireMock.absent())
353 					.willReturn(WireMock.unauthorized().withHeader(HttpHeaders.WWW_AUTHENTICATE, "Basic")));
354 			mock.stubFor(WireMock.any(WireMock.anyUrl())
355 					.withHeader(HttpHeaders.AUTHORIZATION, WireMock.matching(".*"))
356 					.willReturn(WireMock.ok()));
357 			requestPattern = RequestPatternBuilder.newRequestPattern()
358 					.withHeader(HttpHeaders.AUTHORIZATION, WireMock.matching(".*"));
359 		}
360 
361 		assertTrue(path.startsWith("/"), "path should start with slash");
362 		assertDoesNotThrow(() -> mojo.execute(), "upload failed");
363 		List<LoggedRequest> requests = mock.findAll(requestPattern);
364 		assertEquals(1, requests.size(), "expected only one request");
365 		LoggedRequest request = requests.get(0);
366 		assertEquals(method, request.getMethod(), "method");
367 		assertEquals((mojo.getUploadRepoStable().getUrl().startsWith("https")
368 				? "https://127.0.0.1:" + mock.getHttpsPort()
369 				: "http://127.0.0.1:" + mock.getPort()) + path, request.getAbsoluteUrl(), "url");
370 		assertEquals("application/gzip", request.getHeader(HttpHeaders.CONTENT_TYPE), "content-type");
371 		assertEquals(authorization, request.getHeader(HttpHeaders.AUTHORIZATION), "authorization");
372 	}

374 	private void assertCatalog(UploadMojo mojo, Path archive, String uploadUrl) {
375 		File helmCatalogFile = mojo.getCatalogFilePath().toFile();
376 		assertTrue(helmCatalogFile.exists());
377 		List<Catalog> catalogs;
378 		try {
379 			catalogs = mojo.readCatalog(helmCatalogFile);
380 		} catch (MojoExecutionException e) {
381 			throw new RuntimeException(e);
382 		}
383 		for (Catalog catalog : catalogs) {
384 			assertEquals(archive.toString(), catalog.getChart().toString());
385 			assertEquals(uploadUrl, catalog.getUploadUrl().toString());
386 			assertNull(catalog.getUploadResponseType());
387 			assertEquals("", catalog.getUploadResponse());
388 		}
