271 	@Override
272 	public CodegenOperation fromOperation(String path, String httpMethod, Operation source, List<Server> servers) {
273 
274 		var operation = super.fromOperation(path, httpMethod, source, servers);
275 		var produces = operation.produces == null ? List.<Map<String, String>>of() : operation.produces;
276 
277 		// warn if operation has wildcard/range responses
278 
279 		if (operation.responses.stream().allMatch(CodegenResponse::isWildcard)) {
280 			throw new SpecValidationException("Only default response is not supported.");
281 		}
282 		if (operation.responses.stream().anyMatch(CodegenResponse::isRange)) {
283 			throw new SpecValidationException("Range responses are not supported.");
284 		}
285 
286 		// get responses for considering repsonse type
287 
288 		var responses = operation.responses.stream().filter(r -> r.is2xx || r.is3xx).toList();
289 		var responsesCodes = responses.stream().map(r -> Integer.parseInt(r.code)).toList();
290 		var responseGeneric = useGenericResponse;
291 		var dataTypes = responses.stream().map(r -> r.dataType).collect(Collectors.toSet());
292 
293 		if (!responseGeneric && responses.size() > 1) {
294 			LOG.info("operation {} has multiple responses {}, use generic response",
295 					operation.nickname, responsesCodes);
296 			responseGeneric = true;
297 		}
298 
299 		if (!responseGeneric && responses.stream().anyMatch(r -> r.hasHeaders)) {
300 			LOG.info("operation {} has response with header, use generic response", operation.nickname);
301 			responseGeneric = true;
302 		}
303 
304 		if (operation.returnType != null && dataTypes.size() > 1) {
305 			LOG.info("operation {} has multiple responses {} with multiple models {}, erase return type",
306 					operation.nickname, responsesCodes, dataTypes);
307 			operation.returnType = null;
308 		}
309 
310 		if ((operation.returnType != null || !responseGeneric) && produces.size() > 1) {
311 			var specResponse = (ApiResponse) responses.get(0).vendorExtensions.get(ApiResponse.class.getName());
312 			var returnTypes = specResponse.getContent().values().stream().collect(Collectors.toSet());
313 			if (returnTypes.size() > 1) {
314 				var mediaTypes = produces.stream().flatMap(m -> m.values().stream()).collect(Collectors.toSet());
315 				LOG.info("operation {} has multiple media types {} with multiple models, erase return type",
316 						operation.nickname, mediaTypes);
317 				operation.returnType = null;
318 				responseGeneric = true;
319 			}
320 		}
321 
322 		// store method and status for micronaut
323 
324 		var extensions = operation.vendorExtensions;
325 		extensions.put("httpMethod", httpMethod.toUpperCase().charAt(0) + httpMethod.substring(1).toLowerCase());
326 		extensions.put("generic", responseGeneric);
327 		if (responses.size() == 1) {
328 			extensions.put("status", HttpStatus.valueOf(responsesCodes.get(0)).name());
329 		}
330 		operation.responses.forEach(r -> extensions.put("has" + r.code, true));
331 
332 		// jwt provider for tests
333 
334 		var hasSecurityJwt = (boolean) operation.vendorExtensions.getOrDefault("has401", false);
335 		if (generateApiTests && hasSecurityJwt) {
336 			addSupportingFile(testFolder, invokerPackage, "JwtProvider");
337 			addSupportingFile(testFolder, invokerPackage, "JwtBuilder");
338 		}
339 
340 		// async
341 
342 		if (supportAsync) {
343 			var isVoid = operation.returnType == null;
344 			var isStream = Optional.ofNullable(source.getResponses().get("200"))
345 					.map(ApiResponse::getContent)
346 					.filter(content -> content.containsKey(MediaType.APPLICATION_JSON_STREAM))
347 					.isPresent();
348 			extensions.put("asyncContainer", typeMapping.get("asyncSingle"));
349 			extensions.put("asyncStream", isStream);
350 			if (!responseGeneric) {
351 				if (isVoid) {
352 					var asyncCompletable = typeMapping.get("asyncCompletable");
353 					if ("reactor.core.publisher.Mono".equals(asyncCompletable)) {
354 						extensions.put("asyncContainer", asyncCompletable + "<java.lang.Void>");
355 					} else {
356 						extensions.put("asyncContainer", asyncCompletable);
357 					}
358 				} else if ((boolean) operation.vendorExtensions.getOrDefault("has404", false)) {
359 					extensions.put("asyncContainer", typeMapping.get("asyncMaybe"));
360 				}
361 			}
362 			if (isStream) {
363 				extensions.put("asyncContainer", typeMapping.get("asyncFlowable"));
364 			}
365 		}
366 
367 		// add upper case operationId for path constants
368 
369 		operation.vendorExtensions.put("operationIdUpperCase",
370 				StringUtils.underscore(operation.nickname).toUpperCase());
371 
372 		operation.pathParams.stream().filter(p -> p.defaultValue != null).forEach(p -> {
373 			LOG.warn("operation {} has path param {} with unsupported default value {}, default removed",
374 					operation.nickname, p.baseName, p.defaultValue);
375 			p.defaultValue = null;
376 		});
377 
378 		// for handle client/server specific path
379 
380 		var clientPath = operation.path;
381 		var queryParamsWithArray = operation.queryParams.stream().filter(p -> p.isArray).toList();
382 		if (!queryParamsWithArray.isEmpty()) {
383 			clientPath = operation.path + "?"
384 					+ queryParamsWithArray.stream().map(p -> "{&" + p.baseName + "*}").collect(Collectors.joining());
385 		}
386 		extensions.put("clientPath", clientPath);
387 
388 		// for handle client/server specific path
389 
390 		var serverPath = operation.path;
391 		for (var pathParam : operation.pathParams) {
392 			var name = pathParam.baseName;
393 			if (pathParam.isBoolean) {
394 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":true|false}");
395 			} else if (pathParam.isInteger || pathParam.isLong) {
396 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":\\\\-?[0-9]+}");
397 			} else if (pathParam.isUuid) {
398 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":" + UUID_PATTERN + "}");
399 			} else if (pathParam.maxLength != null) {
400 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":" + pathParam.maxLength + "}");
401 			} else if (pathParam.pattern != null && !pathParam.pattern.contains("{")) {
402 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":" + pathParam.pattern + "}");
403 			}
404 		}
405 		extensions.put("serverPath", serverPath);
406 
407 		return operation;
408 	}

422 	@Override
423 	public Map<String, ModelsMap> updateAllModels(Map<String, ModelsMap> objs) {
424 		var superObjs = super.updateAllModels(objs);
425 
426 		var allModels = getAllModels(objs);
427 		for (CodegenModel model : allModels.values()) {
428 
429 			// check if composed schemas for additional properties should be handled and apply to the map if so.
430 
431 			if (supportsAdditionalPropertiesWithComposedSchema && model.getAdditionalProperties() != null) {
432 				model.getVendorExtensions().put("additionalPropertiesMap", Map.of(
433 						"keyType", "java.lang.String",
434 						"valueType", model.getAdditionalProperties().getDataType()));
435 			}
436 
437 			// oneOf handling
438 
439 			if (model.oneOf != null && !model.oneOf.isEmpty()) {
440 				model.children = model.oneOf.stream().map(allModels::get).collect(Collectors.toList());
441 				model.hasChildren = true;
442 				model.interfaceModels.removeAll(model.children);
443 				for (var child : model.children) {
444 					if (child.interfaceModels == null) {
445 						child.interfaceModels = new ArrayList<>();
446 					}
447 					child.interfaceModels.add(model);
448 				}
449 				model.allVars = new ArrayList<>();
450 				model.vars = new ArrayList<>();
451 				model.hasVars = false;
452 			}
453 
454 			// model all parents as interfaces
455 
456 			model.parent = null;
457 			model.parentModel = null;
458 			model.parentSchema = null;
459 			model.parentVars = List.of();
460 			model.allParents = List.of();
461 			if (model.interfaceModels != null) {
462 				for (var interfaceModel : model.interfaceModels) {
463 
464 					// add missing children
465 					if (interfaceModel.children == null) {
466 						interfaceModel.children = new ArrayList<>();
467 					}
468 					if (!interfaceModel.children.contains(model)) {
469 						LOG.debug("Add model " + model.name + " as children to " + interfaceModel.name + ".");
470 						interfaceModel.children.add(model);
471 						interfaceModel.hasChildren = true;
472 					}
473 
474 					// strip parent vars from vars
475 					model.vars.removeIf(property -> {
476 						var remove = interfaceModel.allVars.stream()
477 								.map(CodegenProperty::getBaseName)
478 								.anyMatch(property.getBaseName()::equals);
479 						if (remove) {
480 							LOG.debug("Remove property " + property.getBaseName() + " from " + model.name
481 									+ " because already present on " + interfaceModel.name + ".");
482 						}
483 						return remove;
484 					});
485 				}
486 			}
487 
488 			// handle discriminator
489 
490 			var discriminator = model.discriminator;
491 			if (discriminator == null) {
492 				continue;
493 			}
494 
495 			// remove discriminator type
496 
497 			model.vars.stream()
498 					.filter(property -> property.getName().equals(discriminator.getPropertyName()))
499 					.findAny().ifPresent(property -> {
500 						discriminator.setPropertyType(property.getDataType());
501 						model.vars.remove(property);
502 					});
503 
504 			// add discriminator value to submodel
505 
506 			for (var mappedModel : discriminator.getMappedModels()) {
507 
508 				var subModel = allModels.get(mappedModel.getModelName());
509 				if (subModel == null) {
510 					LOG.warn("{} - model in discriminator {} with name {} not found",
511 							model.name, discriminator.getPropertyName(), mappedModel.getModelName());
512 					continue;
513 				}
514 				subModel.vars.removeIf(property -> property.getName().equals(discriminator.getPropertyName()));
515 				subModel.allVars.removeIf(property -> property.getName().equals(discriminator.getPropertyName()));
516 
517 				var extensions = subModel.vendorExtensions;
518 				extensions.put("discriminatorPropertyGetter", discriminator.getPropertyGetter());
519 				extensions.put("discriminatorPropertyType", discriminator.getPropertyType());
520 				extensions.put("discriminatorPropertyValue", switch (discriminator.getPropertyType()) {
521 					case "java.lang.String" -> '"' + mappedModel.getMappingName() + '"';
522 					case "java.lang.Long", "java.lang.Integer" -> mappedModel.getMappingName();
523 					case "java.lang.Double", "java.lang.Float" -> mappedModel.getMappingName();
524 					default -> discriminator.getPropertyType() + "." + toEnumVarName(mappedModel.getMappingName(), "");
525 				});
526 			}
527 		}
528 
529 		// add default classes
530 
531 		for (var model : allModels.values()) {
532 			if (model.hasChildren && model.discriminator == null && model.oneOf.isEmpty()) {
533 
534 				var suffix = "Default";
535 				var copy = copy(model);
536 				copy.name = model.name + suffix;
537 				copy.classname = model.classname + suffix;
538 				copy.classFilename = model.classFilename + suffix;
539 				copy.children = null;
540 				copy.hasChildren = false;
541 				copy.interfaces = List.of(model.name);
542 				copy.interfaceModels = List.of(model);
543 				copy.vars = List.of();
544 
545 				var obj = new ModelsMap();
546 				obj.putAll(superObjs.get(model.name));
547 				obj.put("classname", copy.classname);
548 				obj.setModels(List.of(new ModelMap(Map.of("model", copy))));
549 				superObjs.put(model.name + "Default", obj);
550 			}
551 
552 		}
553 
554 		return superObjs;
555 	}

589 	@Override
590 	public String toExampleValue(@SuppressWarnings("rawtypes") Schema schema) {
591 
592 		// first choice: use the example, provided from the spec
593 		// second choice: use the default provided by the spec
594 		// special handling for enum: if no example or default is provided, use the first value
595 		@SuppressWarnings("unchecked")
596 		Optional<String> value = Optional
597 				.ofNullable(schema.getExample())
598 				.or(() -> Optional.ofNullable(schema.getDefault()))
599 				.or(() -> Optional.ofNullable(schema.getEnum()).flatMap(e -> e.stream().findFirst()))
600 				.map(Object::toString);
601 
602 		// third choice: set type-specific default examples
603 
604 		if (ModelUtils.isBooleanSchema(schema)) {
605 			return value.orElse("false");
606 		}
607 		if (ModelUtils.isLongSchema(schema)) {
608 			return value.orElse("100L") + "L";
609 		}
610 		if (ModelUtils.isFloatSchema(schema)) {
611 			return value.orElse("12.34") + "F";
612 		}
613 		if (ModelUtils.isDoubleSchema(schema)) {
614 			return value.orElse("43.21") + "D";
615 		}
616 		if (ModelUtils.isIntegerSchema(schema) || ModelUtils.isShortSchema(schema)) {
617 			return value.orElse("12");
618 		}
619 		if (ModelUtils.isNumberSchema(schema)) {
620 			return value.map(v -> "java.lang.Number.valueOf(\"" + v + "\")").orElse("12.34");
621 		}
622 
623 		if (ModelUtils.isDateSchema(schema)) {
624 			if ("java.time.LocalDate".equals(typeMapping.get(schema.getType()))) {
625 				return "java.time.LocalDate." + value.map(v -> "parse(\"" + v + "\")").orElse("now()");
626 			}
627 			// we cannot provide a valid example for all possible date types
628 			return "null";
629 		}
630 		if (ModelUtils.isDateTimeSchema(schema)) {
631 			if ("java.time.Instant".equals(typeMapping.get(schema.getType()))) {
632 				return "java.time.Instant." + value.map(v -> "parse(\"" + v + "\")").orElse("now()");
633 			}
634 			// we cannot provide a valid example for all possible date types
635 			return "null";
636 		}
637 		if (ModelUtils.isByteArraySchema(schema) || ModelUtils.isBinarySchema(schema)) {
638 			return value.orElse("\"byteArray\".getBytes()");
639 		}
640 		if (ModelUtils.isFileSchema(schema)) {
641 			return value.orElse("\"myFile\".getBytes()");
642 		}
643 		if (ModelUtils.isUUIDSchema(schema)) {
644 			return "java.util.UUID." + value.map(v -> "fromString(\"" + v + "\")").orElse("randomUUID()");
645 		}
646 		if (ModelUtils.isURISchema(schema)) {
647 			return "java.net.URI.create(\"" + value.orElse("my:uri") + "\")";
648 		}
649 		if (ModelUtils.isEmailSchema(schema)) {
650 			return '"' + value.orElse("mail@example.org") + '"';
651 		}
652 		if (ModelUtils.isStringSchema(schema)) {
653 			return '"' + value.orElse("string") + '"';
654 		}
655 
656 		if (ModelUtils.isMapSchema(schema)) {
657 			return "java.util.Map." + value
658 					.map(v -> (Map<?, ?>) new org.yaml.snakeyaml.Yaml().loadAs(v, Map.class))
659 					.filter(map -> !map.isEmpty())
660 					.map(map -> map.entrySet().stream()
661 							.map(e -> "new java.util.AbstractMap.SimpleEntry(\""
662 									+ e.getKey() + "\", \"" + e.getValue() + "\")")
663 							.collect(Collectors.joining(", ")))
664 					.map(map -> "ofEntries(" + map + ")")
665 					.orElse("of()");
666 		}
667 		if (ModelUtils.isSet(schema)) {
668 			return "java.util.Set.of(" + value.map(v -> v.substring(1, v.length() - 1)).orElse("") + ")";
669 		}
670 		if (ModelUtils.isArraySchema(schema)) {
671 			return "java.util.List.of(" + value.map(v -> v.substring(1, v.length() - 1)).orElse("") + ")";
672 		}
673 
674 		// if no example can be generated, leave it null.
675 		return "null";
676 	}
