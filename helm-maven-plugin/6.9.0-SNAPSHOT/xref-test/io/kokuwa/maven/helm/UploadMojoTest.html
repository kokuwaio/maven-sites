260 	private void assertUpload(UploadMojo mojo, RequestMethod method, String path, String authorization) {
261 
262 		RequestPatternBuilder requestPattern;
263 		if (authorization == null) {
264 			mock.stubFor(WireMock.any(WireMock.anyUrl()).willReturn(WireMock.ok()));
265 			requestPattern = RequestPatternBuilder.allRequests();
266 		} else {
267 			mock.stubFor(WireMock.any(WireMock.anyUrl())
268 					.withHeader(HttpHeaders.AUTHORIZATION, WireMock.absent())
269 					.willReturn(WireMock.unauthorized().withHeader(HttpHeaders.WWW_AUTHENTICATE, "Basic")));
270 			mock.stubFor(WireMock.any(WireMock.anyUrl())
271 					.withHeader(HttpHeaders.AUTHORIZATION, WireMock.matching(".*"))
272 					.willReturn(WireMock.ok()));
273 			requestPattern = RequestPatternBuilder.newRequestPattern()
274 					.withHeader(HttpHeaders.AUTHORIZATION, WireMock.matching(".*"));
275 		}
276 
277 		assertTrue(path.startsWith("/"), "path should start with slash");
278 		assertDoesNotThrow(() -> mojo.execute(), "upload failed");
279 		List<LoggedRequest> requests = mock.findAll(requestPattern);
280 		assertEquals(1, requests.size(), "expected only one request");
281 		LoggedRequest request = requests.get(0);
282 		assertEquals(method, request.getMethod(), "method");
283 		assertEquals("http://127.0.0.1:" + mock.getPort() + path, request.getAbsoluteUrl(), "url");
284 		assertEquals("application/gzip", request.getHeader(HttpHeaders.CONTENT_TYPE), "content-type");
285 		assertEquals(authorization, request.getHeader(HttpHeaders.AUTHORIZATION), "authorization");
286 	}
