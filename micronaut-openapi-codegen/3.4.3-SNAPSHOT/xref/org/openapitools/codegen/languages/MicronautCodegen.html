268 	@Override
269 	public CodegenOperation fromOperation(String path, String httpMethod, Operation source, List<Server> servers) {
270 
271 		var operation = super.fromOperation(path, httpMethod, source, servers);
272 		var produces = operation.produces == null ? List.<Map<String, String>>of() : operation.produces;
273 
274 		// warn if operation has wildcard/range responses
275 
276 		if (operation.responses.stream().allMatch(CodegenResponse::isWildcard)) {
277 			throw new SpecValidationException("Only default response is not supported.");
278 		}
279 		if (operation.responses.stream().anyMatch(CodegenResponse::isRange)) {
280 			throw new SpecValidationException("Range responses are not supported.");
281 		}
282 
283 		// get responses for considering repsonse type
284 
285 		var responses = operation.responses.stream().filter(r -> r.is2xx || r.is3xx).collect(Collectors.toList());
286 		var responsesCodes = responses.stream().map(r -> Integer.parseInt(r.code)).collect(Collectors.toList());
287 		var responseGeneric = useGenericResponse;
288 		var dataTypes = responses.stream().map(r -> r.dataType).collect(Collectors.toSet());
289 
290 		if (!responseGeneric && responses.size() > 1) {
291 			LOG.info("operation {} has multiple responses {}, use generic response",
292 					operation.nickname, responsesCodes);
293 			responseGeneric = true;
294 		}
295 
296 		if (!responseGeneric && responses.stream().anyMatch(r -> r.hasHeaders)) {
297 			LOG.info("operation {} has response with header, use generic response", operation.nickname);
298 			responseGeneric = true;
299 		}
300 
301 		if (operation.returnType != null && dataTypes.size() > 1) {
302 			LOG.info("operation {} has multiple responses {} with multiple models {}, erase return type",
303 					operation.nickname, responsesCodes, dataTypes);
304 			operation.returnType = null;
305 		}
306 
307 		if ((operation.returnType != null || !responseGeneric) && produces.size() > 1) {
308 			var specResponse = (ApiResponse) responses.get(0).vendorExtensions.get(ApiResponse.class.getName());
309 			var returnTypes = specResponse.getContent().values().stream().collect(Collectors.toSet());
310 			if (returnTypes.size() > 1) {
311 				var mediaTypes = produces.stream().flatMap(m -> m.values().stream()).collect(Collectors.toSet());
312 				LOG.info("operation {} has multiple media types {} with multiple models, erase return type",
313 						operation.nickname, mediaTypes);
314 				operation.returnType = null;
315 				responseGeneric = true;
316 			}
317 		}
318 
319 		// store method and status for micronaut
320 
321 		var extensions = operation.vendorExtensions;
322 		extensions.put("httpMethod", httpMethod.toUpperCase().charAt(0) + httpMethod.substring(1).toLowerCase());
323 		extensions.put("generic", responseGeneric);
324 		if (responses.size() == 1) {
325 			extensions.put("status", HttpStatus.valueOf(responsesCodes.get(0)).name());
326 		}
327 		operation.responses.forEach(r -> extensions.put("has" + r.code, true));
328 
329 		// jwt provider for tests
330 
331 		var hasSecurityJwt = (boolean) operation.vendorExtensions.getOrDefault("has401", false);
332 		if (generateApiTests && hasSecurityJwt) {
333 			addSupportingFile(testFolder, invokerPackage, "JwtProvider");
334 			addSupportingFile(testFolder, invokerPackage, "JwtBuilder");
335 		}
336 
337 		// async
338 
339 		if (supportAsync) {
340 			var isVoid = operation.returnType == null;
341 			var isStream = Optional.ofNullable(source.getResponses().get("200"))
342 					.map(ApiResponse::getContent)
343 					.filter(content -> content.containsKey(MediaType.APPLICATION_JSON_STREAM))
344 					.isPresent();
345 			extensions.put("asyncContainer", typeMapping.get("asyncSingle"));
346 			extensions.put("asyncStream", isStream);
347 			if (!responseGeneric) {
348 				if (isVoid) {
349 					var asyncCompletable = typeMapping.get("asyncCompletable");
350 					if ("reactor.core.publisher.Mono".equals(asyncCompletable)) {
351 						extensions.put("asyncContainer", asyncCompletable + "<java.lang.Void>");
352 					} else {
353 						extensions.put("asyncContainer", asyncCompletable);
354 					}
355 				} else if ((boolean) operation.vendorExtensions.getOrDefault("has404", false)) {
356 					extensions.put("asyncContainer", typeMapping.get("asyncMaybe"));
357 				}
358 			}
359 			if (isStream) {
360 				extensions.put("asyncContainer", typeMapping.get("asyncFlowable"));
361 			}
362 		}
363 
364 		// add upper case operationId for path constants
365 
366 		operation.vendorExtensions.put("operationIdUpperCase",
367 				StringUtils.underscore(operation.nickname).toUpperCase());
368 
369 		operation.pathParams.stream().filter(p -> p.defaultValue != null).forEach(p -> {
370 			LOG.warn("operation {} has path param {} with unsupported default value {}, default removed",
371 					operation.nickname, p.baseName, p.defaultValue);
372 			p.defaultValue = null;
373 		});
374 
375 		// for handle client/server specific path
376 
377 		var clientPath = operation.path;
378 		var queryParamsWithArray = operation.queryParams.stream().filter(p -> p.isArray).collect(Collectors.toList());
379 		if (!queryParamsWithArray.isEmpty()) {
380 			clientPath = operation.path + "?"
381 					+ queryParamsWithArray.stream().map(p -> "{&" + p.baseName + "*}").collect(Collectors.joining());
382 		}
383 		extensions.put("clientPath", clientPath);
384 
385 		// for handle client/server specific path
386 
387 		var serverPath = operation.path;
388 		for (var pathParam : operation.pathParams) {
389 			var name = pathParam.baseName;
390 			if (pathParam.isBoolean) {
391 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":true|false}");
392 			} else if (pathParam.isInteger || pathParam.isLong) {
393 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":\\\\-?[0-9]+}");
394 			} else if (pathParam.isUuid) {
395 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":" + UUID_PATTERN + "}");
396 			} else if (pathParam.maxLength != null) {
397 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":" + pathParam.maxLength + "}");
398 			} else if (pathParam.pattern != null && !pathParam.pattern.contains("{")) {
399 				serverPath = serverPath.replace("{" + name + "}", "{" + name + ":" + pathParam.pattern + "}");
400 			}
401 		}
402 		extensions.put("serverPath", serverPath);
403 
404 		return operation;
405 	}

419 	@Override
420 	public Map<String, ModelsMap> updateAllModels(Map<String, ModelsMap> objs) {
421 		var superObjs = super.updateAllModels(objs);
422 
423 		var allModels = getAllModels(objs);
424 		for (CodegenModel model : allModels.values()) {
425 
426 			// check if composed schemas for additional properties should be handled and apply to the map if so.
427 
428 			if (supportsAdditionalPropertiesWithComposedSchema && model.getAdditionalProperties() != null) {
429 				model.getVendorExtensions().put("additionalPropertiesMap", Map.of(
430 						"keyType", "java.lang.String",
431 						"valueType", model.getAdditionalProperties().getDataType()));
432 			}
433 
434 			// handle discriminator
435 
436 			var discriminator = model.discriminator;
437 			if (discriminator == null) {
438 				continue;
439 			}
440 
441 			// remove discriminator type
442 
443 			model.vars.stream()
444 					.filter(property -> property.getName().equals(discriminator.getPropertyName()))
445 					.findAny().ifPresent(property -> {
446 						discriminator.setPropertyType(property.getDataType());
447 						model.vars.remove(property);
448 					});
449 
450 			// add discriminator value to submodel
451 
452 			for (var mappedModel : discriminator.getMappedModels()) {
453 
454 				var subModel = allModels.get(mappedModel.getModelName());
455 				if (subModel == null) {
456 					LOG.warn("{} - model in discriminator {} with name {} not found",
457 							model.name, discriminator.getPropertyName(), mappedModel.getModelName());
458 					continue;
459 				}
460 				if (subModel.parentModel == null) {
461 					subModel.parentModel = model;
462 					subModel.parent = model.getClassname();
463 					LOG.warn("{} added missing sub model {}", model.name, subModel.name);
464 				}
465 				subModel.vars.removeIf(property -> property.getName().equals(discriminator.getPropertyName()));
466 				subModel.allVars.removeIf(property -> property.getName().equals(discriminator.getPropertyName()));
467 
468 				var extensions = subModel.vendorExtensions;
469 				extensions.put("discriminatorPropertyGetter", discriminator.getPropertyGetter());
470 				extensions.put("discriminatorPropertyType", discriminator.getPropertyType());
471 				switch (discriminator.getPropertyType()) {
472 					case "java.lang.String":
473 						extensions.put("discriminatorPropertyValue", '"' + mappedModel.getMappingName() + '"');
474 						break;
475 					case "java.lang.Long":
476 					case "java.lang.Integer":
477 					case "java.lang.Double":
478 					case "java.lang.Float":
479 						extensions.put("discriminatorPropertyValue", mappedModel.getMappingName());
480 						break;
481 					default:
482 						extensions.put("discriminatorPropertyValue", discriminator.getPropertyType() + "."
483 								+ toEnumVarName(mappedModel.getMappingName(), ""));
484 				}
485 			}
486 		}
487 
488 		return superObjs;
489 	}

523 	@Override
524 	public String toExampleValue(Schema schema) {
525 
526 		// first choice: use the example, provided from the spec
527 		// second choice: use the default provided by the spec
528 		// special handling for enum: if no example or default is provided, use the first value
529 		Optional<String> value = Optional
530 				.ofNullable(schema.getExample())
531 				.or(() -> Optional.ofNullable(schema.getDefault()))
532 				.or(() -> Optional.ofNullable(schema.getEnum()).flatMap(e -> e.stream().findFirst()))
533 				.map(Object::toString);
534 
535 		// third choice: set type-specific default examples
536 
537 		if (ModelUtils.isBooleanSchema(schema)) {
538 			return value.orElse("false");
539 		}
540 		if (ModelUtils.isLongSchema(schema)) {
541 			return value.orElse("100L") + "L";
542 		}
543 		if (ModelUtils.isFloatSchema(schema)) {
544 			return value.orElse("12.34") + "F";
545 		}
546 		if (ModelUtils.isDoubleSchema(schema)) {
547 			return value.orElse("43.21") + "D";
548 		}
549 		if (ModelUtils.isIntegerSchema(schema) || ModelUtils.isShortSchema(schema)) {
550 			return value.orElse("12");
551 		}
552 		if (ModelUtils.isNumberSchema(schema)) {
553 			return value.map(v -> "java.lang.Number.valueOf(\"" + v + "\")").orElse("12.34");
554 		}
555 
556 		if (ModelUtils.isDateSchema(schema)) {
557 			if ("java.time.LocalDate".equals(typeMapping.get(schema.getType()))) {
558 				return "java.time.LocalDate." + value.map(v -> "parse(\"" + v + "\")").orElse("now()");
559 			} else {
560 				// we cannot provide a valid example for all possible date types
561 				return "null";
562 			}
563 		}
564 		if (ModelUtils.isDateTimeSchema(schema)) {
565 			if ("java.time.Instant".equals(typeMapping.get(schema.getType()))) {
566 				return "java.time.Instant." + value.map(v -> "parse(\"" + v + "\")").orElse("now()");
567 			} else {
568 				// we cannot provide a valid example for all possible date types
569 				return "null";
570 			}
571 		}
572 		if (ModelUtils.isByteArraySchema(schema) || ModelUtils.isBinarySchema(schema)) {
573 			return value.orElse("\"byteArray\".getBytes()");
574 		}
575 		if (ModelUtils.isFileSchema(schema)) {
576 			return value.orElse("\"myFile\".getBytes()");
577 		}
578 		if (ModelUtils.isUUIDSchema(schema)) {
579 			return "java.util.UUID." + value.map(v -> "fromString(\"" + v + "\")").orElse("randomUUID()");
580 		}
581 		if (ModelUtils.isURISchema(schema)) {
582 			return "java.net.URI.create(\"" + value.orElse("my:uri") + "\")";
583 		}
584 		if (ModelUtils.isEmailSchema(schema)) {
585 			return '"' + value.orElse("mail@example.org") + '"';
586 		}
587 		if (ModelUtils.isStringSchema(schema)) {
588 			return '"' + value.orElse("string") + '"';
589 		}
590 
591 		if (ModelUtils.isMapSchema(schema)) {
592 			return "java.util.Map." + value
593 					.map(v -> (Map<?, ?>) new org.yaml.snakeyaml.Yaml().loadAs(v, Map.class))
594 					.filter(map -> !map.isEmpty())
595 					.map(map -> map.entrySet().stream()
596 							.map(e -> "new java.util.AbstractMap.SimpleEntry(\""
597 									+ e.getKey() + "\", \"" + e.getValue() + "\")")
598 							.collect(Collectors.joining(", ")))
599 					.map(map -> "ofEntries(" + map + ")")
600 					.orElse("of()");
601 		}
602 		if (ModelUtils.isSet(schema)) {
603 			return "java.util.Set.of(" + value.map(v -> v.substring(1, v.length() - 1)).orElse("") + ")";
604 		}
605 		if (ModelUtils.isArraySchema(schema)) {
606 			return "java.util.List.of(" + value.map(v -> v.substring(1, v.length() - 1)).orElse("") + ")";
607 		}
608 
609 		// if no example can be generated, leave it null.
610 		return "null";
611 	}
