98  	static void generate(CodegenConfigurator configurator) {
99  		var gen = new DefaultGenerator();
100 		gen.setGenerateMetadata(false);
101 		gen.setGeneratorPropertyDefault(CodegenConstants.SUPPORTING_FILES, "false");
102 		gen.setGeneratorPropertyDefault(CodegenConstants.APIS, "true");
103 		gen.setGeneratorPropertyDefault(CodegenConstants.MODELS, "true");
104 		gen.opts(configurator.toClientOptInput()).generate();
105 	}
