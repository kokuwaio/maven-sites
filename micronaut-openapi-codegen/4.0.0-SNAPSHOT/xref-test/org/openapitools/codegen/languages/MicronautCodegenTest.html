162 	static void generate(CodegenConfigurator configurator) {
163 		var gen = new DefaultGenerator();
164 		gen.setGenerateMetadata(false);
165 		gen.setGeneratorPropertyDefault(CodegenConstants.SUPPORTING_FILES, "false");
166 		gen.setGeneratorPropertyDefault(CodegenConstants.APIS, "true");
167 		gen.setGeneratorPropertyDefault(CodegenConstants.MODELS, "true");
168 
169 		// server
170 
171 		gen.opts(configurator.toClientOptInput()).generate();
172 
173 		// client without model/supporting files
174 
175 		gen.opts(configurator.addAdditionalProperty(MicronautCodegen.CLIENT_ID, "id").toClientOptInput()).generate();
176 	}
