10  @DisplayName("helm:install")
11  public class InstallMojoTest extends AbstractMojoTest {
12  
13  	@DisplayName("default values")
14  	@Test
15  	void install(InstallMojo mojo) {
16  		mojo.setSkipInstall(false);
17  		assertHelm(mojo, "install simple src/test/resources/simple");
18  	}
19  
20  	@DisplayName("with flag skip")
21  	@Test
22  	void skip(InstallMojo mojo) {
23  		assertHelm(mojo.setSkipInstall(false).setSkip(true));
24  		assertHelm(mojo.setSkipInstall(true).setSkip(false));
25  		assertHelm(mojo.setSkipInstall(true).setSkip(true));
26  	}
27  
28  	@DisplayName("with values overrides")
29  	@Test
30  	void valuesFile(InstallMojo mojo) {
31  		mojo.setSkipInstall(false);
32  		mojo.setValues(new ValueOverride().setYamlFile("bar.yaml"));
33  		assertHelm(mojo, "install simple src/test/resources/simple --values bar.yaml");
34  	}
35  
36  	@DisplayName("with dependencies")
37  	@Test
38  	void dependencies(InstallMojo mojo) {
39  		mojo.setSkipInstall(false);
40  		mojo.setChartDirectory(new File("src/test/resources/dependencies"));
41  		assertHelm(mojo,
42  				"install b src/test/resources/dependencies/b",
43  				"install a2 src/test/resources/dependencies/a2",
44  				"install a1 src/test/resources/dependencies/a1",
45  				"install dependencies src/test/resources/dependencies");
46  	}
47  }
