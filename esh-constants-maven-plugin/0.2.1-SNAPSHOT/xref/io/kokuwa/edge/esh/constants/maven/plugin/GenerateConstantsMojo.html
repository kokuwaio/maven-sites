47  /**
48   * Mojo for generating constant class from "ESH-INF" folder. The plugin scans for XML files in the input folder and
49   * looks for a couple of XPath matching nodes to extract constants strings and writes them to a Java constants class
50   * using a Freemarker template.
51   *
52   * @author Fabian Schlegel
53   * @since 0.1.0
54   */
55  @Mojo(name = "generate-esh-constants", defaultPhase = LifecyclePhase.GENERATE_SOURCES, threadSafe = true)
56  public class GenerateConstantsMojo extends AbstractMojo {
57

60  	 */
61  	private static final Set<String> STANDARD_PROPERTIES = Set.of(
62  			/* the key for the vendor property */
63  			"vendor",
64  			/* the key for the model ID property */
65  			"modelId",
66  			/* the key for the serial number property */
67  			"serialNumber",
68  			/* the key for the hardware version property */
69  			"hardwareVersion",
70  			/* the key for the firmware version property */
71  			"firmwareVersion",
72  			/* the key for the MAC address property */
73  			"macAddress");

120 	@Override
121 	public void execute() throws MojoExecutionException, MojoFailureException {
122 
123 		// Create output directory
124 		createOutputDirectory();
125 
126 		// Get input file list
127 		final Set<Path> inputFiles = findXMLFiles();
128 
129 		List<String> bindingIDs = scanDocuments(inputFiles, "//thing-descriptions/@bindingId");
130 		List<String> properties = scanDocuments(inputFiles, "//property/@name");
131 		List<String> thingTypeIDs = scanDocuments(inputFiles, "//thing-type/@id");
132 		List<String> bridgeTypeIDs = scanDocuments(inputFiles, "//bridge-type/@id");
133 		List<String> channelIDs = scanDocuments(inputFiles, "//channel/@id");
134 		List<String> channelTypeIDs = scanDocuments(inputFiles, "//channel-type/@id");
135 		List<String> channelGroupTypeIDs = scanDocuments(inputFiles, "//channel-group-type/@id");
136 		List<String> channelGroupIDs = scanDocuments(inputFiles, "//channel-group/@id");
137 
138 		properties.removeAll(STANDARD_PROPERTIES);
139 
140 		if (bindingIDs.size() != 1) {
141 			throw new MojoFailureException("Expected exactly one binding ID. Please have a look at your XML files!");
142 		}
143 
144 		String bindingId = bindingIDs.iterator().next();
145 
146 		TypeSpec stringConstantsClass;
147 		try {
148 			stringConstantsClass = TypeSpec.classBuilder(stringsClassName)
149 					.addModifiers(PUBLIC, FINAL)
150 					.addMethod(MethodSpec.constructorBuilder().addModifiers(PRIVATE).build())
151 					.addField(FieldSpec
152 									.builder(String.class, "BINDING_ID", PUBLIC, STATIC, FINAL)
153 									.initializer("$S", bindingId)
154 									.build())
155 					.addFields(mapConstants(properties, "PROPERTY_"))
156 					.addFields(mapConstants(thingTypeIDs.stream().map(type -> bindingId + ":" + type)
157 									.collect(Collectors.toList()), "UID_"))
158 					.addFields(mapConstants(bridgeTypeIDs.stream().map(type -> bindingId + ":" + type)
159 									.collect(Collectors.toList()), "UID_"))
160 					.addFields(mapConstants(thingTypeIDs, "THING_TYPE_ID_"))
161 					.addFields(mapConstants(bridgeTypeIDs, "BRIDGE_TYPE_ID_"))
162 					.addFields(deprecated(mapConstants(channelIDs, "CID_")))
163 					.addFields(deprecated(mapConstants(channelTypeIDs, "CHANNEL_TYPE_ID_")))
164 					.addFields(deprecated(mapConstants(channelGroupTypeIDs, "CHANNEL_GROUP_TYPE_ID_")))
165 					.addFields(deprecated(mapConstants(channelGroupIDs, "GID_")))
166 					.addFields(mapConstants(scanDocuments(inputFiles,
167 							"//bridge-type/channels/channel/@id | //thing-type/channels/channel/@id"), "GCID_"))
168 					// add a constant for each channel which is in a group
169 					.addFields(mapConstants(channelGroupTypeIDs.stream().flatMap(
170 									groupType -> scanDocuments(inputFiles,
171 											"//channel-group-type[@id='" + groupType + "']//channel/@id")
172 											.stream()
173 											.flatMap(groupTypeChannel -> scanDocuments(
174 													inputFiles,
175 													"//channel-group[@typeId='" + groupType + "']/@id")
176 													.stream()
177 													.map(group -> group + "#" + groupTypeChannel)))
178 							.collect(Collectors.toCollection(LinkedHashSet::new)), "GCID_"))
179 
180 					.build();
181 		} catch (RuntimeException exception) {
182 			throw new MojoExecutionException("Can't generate string constants class", exception);
183 		}
184 
185 		try {
186 			JavaFile.builder(packageName, stringConstantsClass)
187 					.build()
188 					.writeTo(Path.of(outputDirectory));
189 		} catch (IOException e) {
190 			throw new MojoExecutionException("Can't write string constants output file to " + outputDirectory, e);
191 		}
192 
193 		TypeSpec openhabConstantsClass = TypeSpec.classBuilder(openhabClassName)
194 				.addModifiers(PUBLIC, FINAL)
195 				.addMethod(MethodSpec.constructorBuilder().addModifiers(PRIVATE).build())
196 				.addFields(bridgeTypeIDs.stream()
197 						.map(bridgeTypeId -> thingTypeUidSpec(bridgeTypeId, "BRIDGE_TYPE_ID_"))
198 						.collect(Collectors.toList()))
199 				.addFields(thingTypeIDs.stream()
200 						.map(thingTypeId -> thingTypeUidSpec(thingTypeId, "THING_TYPE_ID_"))
201 						.collect(Collectors.toList()))
202 				.build();
203 
204 		try {
205 			JavaFile.builder(packageName, openhabConstantsClass)
206 					.addStaticImport(ClassName.get(packageName, stringConstantsClass.name), "*")
207 					.build()
208 					.writeTo(Path.of(outputDirectory));
209 		} catch (IOException e) {
210 			throw new MojoExecutionException("Can't write openhab constants output file to " + outputDirectory, e);
211 		}
212 	}

227 	/**
228 	 * Find all XML files in the input folder.
229 	 *
230 	 * @return Set of {@link Path} to XML files
231 	 * @throws MojoExecutionException on error, execution fails
232 	 */
233 	private Set<Path> findXMLFiles() throws MojoExecutionException {
234 		Set<Path> result;
235 		// Recursively walk down the directory tree and put all ".xml" files in the set
236 		try (Stream<Path> walk = Files.walk(Path.of(inputDirectory))) {
237 			result = walk.filter(p -> p.getFileName().toString().endsWith("xml")).collect(Collectors.toSet());
238 		} catch (IOException e) {
239 			throw new MojoExecutionException("Unable to get input files.", e);
240 		}
241 		return result;
242 	}

244 	/**
245 	 * Retrieve strings matching the XPath from the given XML files.
246 	 *
247 	 * @param inputFiles input file list
248 	 * @param expression the XPath expression to scan
249 	 * @return A {@link Set} of strings matching the XPath
250 	 * @throws MojoExecutionException on error, execution fails
251 	 */
252 	private List<String> scanDocuments(Set<Path> inputFiles, String expression) {
253 		final Set<String> result = new LinkedHashSet<>();
254 		for (Path inputFile : inputFiles) {
255 			result.addAll(scanDocument(inputFile.toFile(), expression));
256 		}
257 		return result.stream().sorted().distinct().collect(Collectors.toList());
258 	}
259 
260 	/**
261 	 * Retrieve strings matching the XPath from the given XML file.
262 	 *
263 	 * @param file       the input file
264 	 * @param expression the XPath expression to scan
265 	 * @return A {@link Set} of strings matching the XPath
266 	 * @throws MojoExecutionException on error, execution fails
267 	 */
268 	private Set<String> scanDocument(File file, String expression) {
269 
270 		try {
271 			final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
272 			final DocumentBuilder builder = factory.newDocumentBuilder();
273 			final Document document = builder.parse(file);
274 
275 			final XPathFactory xPathFactory = XPathFactory.newInstance();
276 			final XPath xpath = xPathFactory.newXPath();
277 
278 			final Set<String> result = new LinkedHashSet<>();
279 
280 			// Find node list in document matching the xpath expression
281 			XPathExpression xPathExpression = xpath.compile(expression);
282 			NodeList nodeList = (NodeList) xPathExpression.evaluate(document, XPathConstants.NODESET);
283 			for (int i = 0; i < nodeList.getLength(); i++) {
284 				Node node = nodeList.item(i);
285 				// Put the text content of the node to the map
286 				result.add(node.getTextContent());
287 			}
288 			return result;
289 		} catch (ParserConfigurationException | IOException | SAXException | XPathExpressionException e) {
290 			throw new RuntimeException("Unable to parse XML document: " + file, e);
291 		}
292 	}

294 	private String safeUpper(String name) {
295 		return name.replaceAll("\\W", "_").toUpperCase(Locale.ROOT);
296 	}
297 
298 	private FieldSpec stringConstant(String prefix, String name) {
299 		return FieldSpec
300 				.builder(String.class, prefix + safeUpper(name), PUBLIC, STATIC, FINAL)
301 				.initializer("$S", name)
302 				.build();
303 	}
304 
305 	private List<FieldSpec> mapConstants(Collection<String> input, String prefix) {
306 		return input.stream().map(p -> stringConstant(prefix, p)).collect(Collectors.toList());
307 	}
308 
309 	/**
310 	 * {@code List<FieldSpec>} decorator that adds a @{@link Deprecated} annotation to all fields.
311 	 */
312 	private List<FieldSpec> deprecated(List<FieldSpec> delegates) {
313 		return delegates.stream()
314 				.map(field ->
315 						field.toBuilder()
316 								.addAnnotation(AnnotationSpec.builder(Deprecated.class)
317 								.addMember("forRemoval", "true").build()).build())
318 				.collect(Collectors.toList());
319 	}
320 
321 	private FieldSpec thingTypeUidSpec(String id, String prefix) {
322 		String safeId = safeUpper(id);
323 		return FieldSpec
324 				.builder(THING_TYPE_UID_CLASS_NAME, safeId + "_THING_TYPE_UID", PUBLIC, STATIC, FINAL)
325 				.initializer("new $T(BINDING_ID, $L$L)", THING_TYPE_UID_CLASS_NAME, prefix, safeId)
326 				.build();
327 	}
