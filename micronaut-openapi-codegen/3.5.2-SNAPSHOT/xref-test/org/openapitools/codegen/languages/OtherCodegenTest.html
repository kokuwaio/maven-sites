53  	private static void generate(Class<? extends DefaultCodegen> codegen, String spec, String packageName,
54  			Consumer<CodegenConfigurator> configure) {
55  		var suppressAllWarnings = "@" + SuppressWarnings.class.getName() + "(\"all\")";
56  		var configurator = configurator(spec, packageName)
57  				.setGeneratorName(getName(codegen))
58  				.setInvokerPackage(packageName)
59  				.setModelPackage(packageName)
60  				.setApiPackage(packageName)
61  				.addAdditionalProperty(CodegenConstants.HIDE_GENERATION_TIMESTAMP, true)
62  				.addAdditionalProperty(CodegenConstants.LEGACY_DISCRIMINATOR_BEHAVIOR, true)
63  				.addAdditionalProperty(CodegenConstants.USE_ONEOF_DISCRIMINATOR_LOOKUP, true)
64  				.addAdditionalProperty(JavaMicronautAbstractCodegen.OPT_REACTIVE, false)
65  				.addAdditionalProperty(JavaMicronautAbstractCodegen.OPT_GENERATE_SWAGGER_ANNOTATIONS, false)
66  				.addAdditionalProperty(AbstractJavaCodegen.ADDITIONAL_MODEL_TYPE_ANNOTATIONS, suppressAllWarnings)
67  				.addAdditionalProperty(AbstractJavaCodegen.ADDITIONAL_ENUM_TYPE_ANNOTATIONS, suppressAllWarnings)
68  				.addAdditionalProperty(AbstractJavaCodegen.DATE_LIBRARY, "java8");
69  		configure.accept(configurator);
70  		var gen = new DefaultGenerator();
71  		gen.setGeneratorPropertyDefault(CodegenConstants.SUPPORTING_FILES, "false");
72  		gen.setGeneratorPropertyDefault(CodegenConstants.MODEL_DOCS, "false");
73  		gen.setGeneratorPropertyDefault(CodegenConstants.MODEL_TESTS, "false");
74  		gen.setGeneratorPropertyDefault(CodegenConstants.MODELS, "true");
75  		gen.setGeneratorPropertyDefault(CodegenConstants.API_DOCS, "false");
76  		gen.setGeneratorPropertyDefault(CodegenConstants.API_TESTS, "false");
77  		gen.setGeneratorPropertyDefault(CodegenConstants.APIS, "true");
78  		gen.setGenerateMetadata(false);
79  		gen.opts(configurator.toClientOptInput()).generate();
80  	}
