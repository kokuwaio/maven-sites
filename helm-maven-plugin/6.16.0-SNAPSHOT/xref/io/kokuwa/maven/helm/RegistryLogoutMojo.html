1   package io.kokuwa.maven.helm;
2   
3   import org.apache.maven.plugin.MojoExecutionException;
4   import org.apache.maven.plugins.annotations.LifecyclePhase;
5   import org.apache.maven.plugins.annotations.Mojo;
6   import org.apache.maven.plugins.annotations.Parameter;
7   
8   import io.kokuwa.maven.helm.pojo.HelmRepository;
9   import lombok.Setter;
10  
11  /**
12   * Mojo for executing "helm registry logout".
13   *
14   * @see <a href="https://helm.sh/docs/helm/helm_registry_logout">helm registry logout</a>
15   * @since 6.7.0
16   */
17  @Mojo(name = "registry-logout", defaultPhase = LifecyclePhase.DEPLOY, threadSafe = true)
18  @Setter
19  public class RegistryLogoutMojo extends AbstractHelmMojo {
20  
21  	/**
22  	 * Set this to <code>true</code> to skip invoking registry-logout goal.
23  	 *
24  	 * @since 6.7.0
25  	 */
26  	@Parameter(property = "helm.registry-logout.skip", defaultValue = "false")
27  	private boolean skipRegistryLogout;
28  
29  	@Override
30  	public void execute() throws MojoExecutionException {
31  
32  		if (skip || skipRegistryLogout) {
33  			getLog().info("Skip registry logout");
34  			return;
35  		}
36  
37  		HelmRepository repository = getHelmUploadRepo();
38  		if (repository == null) {
39  			getLog().warn("No upload repo found for logout, skipped.");
40  			return;
41  		}
42  
43  		helm()
44  				.arguments("registry", "logout", repository.getUrl())
45  				.execute("Failed to logout from repository " + repository.getName() + " at " + repository.getUrl());
46  	}
47  }
