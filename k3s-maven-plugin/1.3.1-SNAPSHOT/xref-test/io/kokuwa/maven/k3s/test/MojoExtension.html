23  
24  import io.kokuwa.maven.k3s.mojo.K3sMojo;
25  import io.kokuwa.maven.k3s.util.DebugLog;
26  import io.kokuwa.maven.k3s.util.Docker;
27  
28  /**
29   * JUnit extension to provide prepared mojos as test parameter.
30   *
31   * @author stephan.schnabel@posteo.de
32   */
33  public class MojoExtension implements ParameterResolver, BeforeAllCallback {
34  
35  	private static final String containerName = "k3s-maven-plugin";
36  	private static final String volumeName = "k3s-maven-plugin-junit";
37  	private static final Log log = new DebugLog(new SystemStreamLog(), false);
38  	private static final Docker docker = new Docker(containerName, volumeName, log);
39  	private static final Set<MojoDescriptor> mojos = new HashSet<>();
40  
41  	@Override
42  	public void beforeAll(ExtensionContext context) throws Exception {
43  		if (mojos.isEmpty()) {
44  			var inputStream = K3sMojo.class.getResourceAsStream("/META-INF/maven/plugin.xml");
45  			assertNotNull(inputStream, "Plugin descriptor for not found, run 'mvn plugin:descriptor'.");
46  			new PluginDescriptorBuilder()
47  					.build(new InterpolationFilterReader(new XmlStreamReader(inputStream),
48  							Map.of("project.build.directory", "target", "project.basedir", ".")))
49  					.getMojos()
50  					.forEach(mojos::add);
51  		}
52  	}
53  
54  	@Override
55  	public boolean supportsParameter(ParameterContext parameterContext, ExtensionContext context) {
56  		var type = parameterContext.getParameter().getType();
57  		return mojos.stream().map(MojoDescriptor::getImplementation).anyMatch(type.getName()::equals)
58  				|| type.equals(Log.class)
59  				|| type.equals(Docker.class);
60  	}
61  
62  	@Override
63  	public Object resolveParameter(ParameterContext parameterContext, ExtensionContext context) {
64  
65  		var type = parameterContext.getParameter().getType();
66  		if (type.equals(Log.class)) {
67  			return log;
68  		}
69  		if (type.equals(Docker.class)) {
70  			return docker;
71  		}
72  
73  		var descriptor = mojos.stream().filter(m -> m.getImplementation().equals(type.getName())).findAny().get();
74  		try {
75  			var mojo = (K3sMojo) type.getDeclaredConstructor().newInstance();
76  			for (var parameter : descriptor.getParameters()) {
77  				if (parameter.getDefaultValue() != null) {
78  					var setter = ReflectionUtils.getSetter(parameter.getName(), mojo.getClass());
79  					var parameterType = ReflectionUtils.getSetterType(setter);
80  					if (String.class.equals(parameterType)) {
81  						setter.invoke(mojo, parameter.getDefaultValue());
82  					} else if (File.class.equals(parameterType)) {
83  						setter.invoke(mojo, new File(parameter.getDefaultValue()));
84  					} else if (boolean.class.equals(parameterType)) {
85  						setter.invoke(mojo, Boolean.valueOf(parameter.getDefaultValue()));
86  					} else if (int.class.equals(parameterType)) {
87  						setter.invoke(mojo, Integer.valueOf(parameter.getDefaultValue()));
88  					} else {
89  						fail(parameter.getName() + " has unknown type: " + type);
90  					}
91  				}
92  			}
93  			mojo.setLog(log);
94  			mojo.setContainerName(containerName);
95  			mojo.setVolumeName(volumeName);
96  			return mojo;
97  		} catch (ReflectiveOperationException e) {
98  			throw new ParameterResolutionException("Failed to setup mojo " + descriptor + ".", e);
99  		}
100 	}
101 }
