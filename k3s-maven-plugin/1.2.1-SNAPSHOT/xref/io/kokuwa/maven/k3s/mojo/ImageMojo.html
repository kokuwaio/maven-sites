29  /**
30   * Import images into k3s containerd.
31   *
32   * @since 0.3.0
33   */
34  @Mojo(name = "image", defaultPhase = LifecyclePhase.PRE_INTEGRATION_TEST, requiresProject = false)
35  public class ImageMojo extends K3sMojo {
36  
37  	/**
38  	 * Download given images via "ctr image pull" inside k3s container.
39  	 *
40  	 * @since 0.3.0
41  	 */
42  	@Parameter(property = "k3s.ctrImages")
43  	private Set<String> ctrImages = new HashSet<>();
44  
45  	/**
46  	 * Import given tar files as images via "ctr image import" inside k3s container.
47  	 *
48  	 * @since 0.3.0
49  	 */
50  	@Parameter(property = "k3s.tarFiles")
51  	private Set<Path> tarFiles = new HashSet<>();
52  
53  	/**
54  	 * Copy given images from docker deamon via "ctr image import" inside k3s container.
55  	 *
56  	 * @since 0.3.0
57  	 */
58  	@Parameter(property = "k3s.dockerImages")
59  	private Set<String> dockerImages = new HashSet<>();
60  
61  	/**
62  	 * Always pull docker images or only if not present.
63  	 *
64  	 * @since 0.10.0
65  	 */
66  	@Parameter(property = "k3s.dockerPullAlways", defaultValue = "false")
67  	private boolean dockerPullAlways;
68  
69  	/**
70  	 * Timout for "ctr image pull" or "docker pull" in seconds.
71  	 *
72  	 * @since 0.4.0
73  	 */
74  	@Parameter(property = "k3s.pullTimeout", defaultValue = "1200")
75  	private Duration pullTimeout;
76  
77  	/**
78  	 * Skip starting of k3s container.
79  	 *
80  	 * @since 0.3.0
81  	 */
82  	@Parameter(property = "k3s.skipImage", defaultValue = "false")
83  	private boolean skipImage;
84  
85  	@Override
86  	public void execute() throws MojoExecutionException {
87  
88  		// skip if no image is request
89  
90  		if (isSkip(skipImage) || ctrImages.isEmpty() && tarFiles.isEmpty() && dockerImages.isEmpty()) {
91  			return;
92  		}
93  
94  		// verify container
95  
96  		if (getDocker().getContainer().isEmpty()) {
97  			throw new MojoExecutionException("No k3s container found");
98  		}
99  
100 		// get callables that handle images
101 
102 		var existingImages = getCtrImages();
103 		var tasks = new HashSet<Callable<Boolean>>();
104 		dockerImages.forEach(requestedImage -> tasks.add(() -> docker(existingImages, requestedImage)));
105 		tarFiles.forEach(tarFile -> tasks.add(() -> tar(existingImages, tarFile)));
106 		ctrImages.forEach(requestedImage -> tasks.add(() -> ctr(existingImages, requestedImage)));
107 
108 		// execute callables
109 
110 		try {
111 			var success = true;
112 			for (var future : Executors.newWorkStealingPool().invokeAll(tasks)) {
113 				success &= future.get();
114 			}
115 			if (!success) {
116 				throw new MojoExecutionException("Failed to handle images, see previous log");
117 			}
118 		} catch (InterruptedException | ExecutionException e) {
119 			throw new MojoExecutionException("Failed to handle images", e);
120 		}
121 	}

123 	private boolean tar(Map<String, Map<String, ?>> existingImages, Path tarFile) {
124 
125 		if (!Files.isRegularFile(tarFile)) {
126 			getLog().error("Tar not found: " + tarFile);
127 			return false;
128 		}
129 
130 		try {
131 
132 			// skip if image is already present in ctr
133 
134 			var labelPath = "k3s-maven-tar-path";
135 			var labelChecksum = "k3s-maven-tar-checksum";
136 			var checksum = new Adler32();
137 			checksum.update(Files.readAllBytes(tarFile));
138 			var newChecksum = "adler32:" + checksum.getValue();
139 			var oldChecksum = existingImages.values().stream()
140 					.filter(l -> Optional.ofNullable(l.get(labelPath)).map(tarFile.toString()::equals).orElse(false))
141 					.map(l -> l.get(labelChecksum)).filter(Objects::nonNull)
142 					.findAny().orElse(null);
143 			if (oldChecksum == null) {
144 				getLog().debug("Tar " + tarFile + " does not exists in ctr.");
145 			} else if (oldChecksum.equals(newChecksum)) {
146 				getLog().info("Tar " + tarFile + " present in ctr with checksum " + newChecksum + ", skip.");
147 				return true;
148 			} else {
149 				getLog().debug("Tar " + tarFile + " present in ctr with checksum " + oldChecksum + ", new is: "
150 						+ newChecksum);
151 			}
152 
153 			// import tar into ctr
154 
155 			var destination = Paths.get("/tmp").resolve(tarFile.getFileName() + "_" + System.nanoTime());
156 			var outputPattern = Pattern.compile("^unpacking (?<image>.*) \\(sha256:[0-9a-f]{64}\\).*$");
157 
158 			getDocker().copyToContainer(tarFile, destination);
159 			for (var output : getDocker().exec(pullTimeout, "ctr", "image", "import", destination.toString())) {
160 				var matcher = outputPattern.matcher(output);
161 				if (matcher.matches()) {
162 					getDocker().exec("ctr", "image", "label", matcher.group("image"), labelPath + "=" + tarFile);
163 					getDocker().exec("ctr", "image", "label", matcher.group("image"),
164 							labelChecksum + "=" + newChecksum);
165 				} else {
166 					getLog().warn("Tar " + tarFile + " import output cannot be parsed: " + output);
167 				}
168 			}
169 		} catch (MojoExecutionException | IOException e) {
170 			getLog().error("Failed to import tar: " + tarFile, e);
171 			return false;
172 		}
173 
174 		getLog().info("Imported tar from " + tarFile);
175 		return true;
176 	}

178 	private boolean ctr(Map<String, Map<String, ?>> existingImages, String image) throws MojoExecutionException {
179 
180 		if (existingImages.containsKey(image)) {
181 			getLog().debug("Image " + image + " found in ctr, skip pulling");
182 			return true;
183 		}
184 
185 		getLog().info("Image " + image + " not found, start pulling");
186 		// use crictl instead of cri, because crictl honors custom registry.yaml
187 		// see https://github.com/k3s-io/k3s/issues/5277
188 		getDocker().exec(pullTimeout, "crictl", "pull", image);
189 		getLog().info("Image " + image + " pulled");
190 
191 		return true;
192 	}

194 	private boolean docker(Map<String, Map<String, ?>> existingImages, String image) throws MojoExecutionException {
195 
196 		// pull image
197 
198 		var digest = getDocker().getImage(image).map(ContainerImage::getDigest).orElse(null);
199 		if (dockerPullAlways || digest == null) {
200 			if (digest != null) {
201 				getLog().debug("Image " + image + " found in docker, pull always ...");
202 			} else {
203 				getLog().debug("Image " + image + " not found in docker, pulling ...");
204 			}
205 			try {
206 				getDocker().pullImage(image, pullTimeout);
207 			} catch (MojoExecutionException e) {
208 				getLog().error("Failed to pull docker image " + image, e);
209 				return false;
210 			}
211 			digest = getDocker().getImage(image).map(ContainerImage::getDigest).orElse(null);
212 		} else {
213 			getLog().debug("Image " + image + " found in docker");
214 		}
215 
216 		// skip if image is already present in ctr
217 
218 		var normalizedImage = getDocker().normalizeImage(image);
219 		var label = "k3s-maven-digest";
220 		var oldDigest = existingImages.getOrDefault(normalizedImage, Map.of()).get(label);
221 		if (oldDigest == null) {
222 			getLog().debug("Image " + image + " does not exists in ctr.");
223 		} else if (oldDigest.equals(digest)) {
224 			getLog().info("Image " + image + " present in ctr with digest " + digest + ", skip.");
225 			return true;
226 		} else {
227 			getLog().debug(
228 					"Image " + image + " present in ctr with digest " + oldDigest + ", new digest is: " + digest);
229 		}
230 
231 		// move from docker to ctr
232 
233 		var filename = Paths.get(image.hashCode() + ".tar");
234 		var source = Paths.get(System.getProperty("java.io.tmpdir")).resolve(filename);
235 		var destination = Paths.get("/tmp").resolve(filename);
236 		try {
237 			getDocker().saveImage(image, source);
238 			getDocker().copyToContainer(source, destination);
239 			getDocker().exec(pullTimeout, "ctr", "image", "import", destination.toString());
240 			getDocker().exec("ctr", "image", "label", normalizedImage, label + "=" + digest);
241 		} catch (MojoExecutionException e) {
242 			getLog().error("Failed to import tar " + source, e);
243 			return false;
244 		}
245 
246 		getLog().info("Image " + image + " copied from docker deamon");
247 		return true;
248 	}

250 	/**
251 	 * Read image from <code>ctr image list</code>.
252 	 *
253 	 * @param row Row from ctr output.
254 	 * @return Image name with labels.
255 	 */
256 	private Map<String, Map<String, ?>> getCtrImages() throws MojoExecutionException {
257 		return getDocker().exec("ctr", "image", "list").stream().map(String::strip)
258 				.filter(row -> !row.startsWith("REF") && !row.startsWith("sha256:"))
259 				.map(row -> row.split("(\\s)+"))
260 				.filter(parts -> {
261 					var matches = parts.length == 7;
262 					if (!matches) {
263 						getLog().warn("Unexpected output of `ctr image list`: " + List.of(parts));
264 					}
265 					return matches;
266 				})
267 				.map(parts -> Map.entry(parts[0], Stream.of(parts[6].split(",")).map(s -> s.split("="))
268 						.filter(s -> !"io.cri-containerd.image".equals(s[0]))
269 						.collect(Collectors.toMap(s -> s[0], s -> s[1]))))
270 				.peek(entry -> getLog().debug("Found ctr image: " + entry))
271 				.collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));
272 	}
