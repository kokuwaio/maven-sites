407 	private void assertUpload(UploadMojo mojo, RequestMethod method, String path, String authorization) {
408 
409 		RequestPatternBuilder requestPattern;
410 		if (authorization == null) {
411 			mock.stubFor(WireMock.any(WireMock.anyUrl()).willReturn(WireMock.ok()));
412 			requestPattern = RequestPatternBuilder.allRequests();
413 		} else {
414 			mock.stubFor(WireMock.any(WireMock.anyUrl())
415 					.withHeader(HttpHeaders.AUTHORIZATION, WireMock.absent())
416 					.willReturn(WireMock.unauthorized().withHeader(HttpHeaders.WWW_AUTHENTICATE, "Basic")));
417 			mock.stubFor(WireMock.any(WireMock.anyUrl())
418 					.withHeader(HttpHeaders.AUTHORIZATION, WireMock.matching(".*"))
419 					.willReturn(WireMock.ok()));
420 			requestPattern = RequestPatternBuilder.newRequestPattern()
421 					.withHeader(HttpHeaders.AUTHORIZATION, WireMock.matching(".*"));
422 		}
423 
424 		assertTrue(path.startsWith("/"), "path should start with slash");
425 		assertDoesNotThrow(() -> mojo.execute(), "upload failed");
426 		List<LoggedRequest> requests = mock.findAll(requestPattern);
427 		assertEquals(1, requests.size(), "expected only one request");
428 		LoggedRequest request = requests.get(0);
429 		assertEquals(method, request.getMethod(), "method");
430 		assertEquals((mojo.getUploadRepoStable().getUrl().startsWith("https")
431 				? "https://127.0.0.1:" + mock.getHttpsPort()
432 				: "http://127.0.0.1:" + mock.getPort()) + path, request.getAbsoluteUrl(), "url");
433 		assertEquals("application/gzip", request.getHeader(HttpHeaders.CONTENT_TYPE), "content-type");
434 		assertEquals(authorization, request.getHeader(HttpHeaders.AUTHORIZATION), "authorization");
435 	}

437 	private void assertCatalog(UploadMojo mojo, Path archive, String uploadUrl) {
438 		File helmCatalogFile = mojo.getCatalogFilePath().toFile();
439 		assertTrue(helmCatalogFile.exists());
440 		try {
441 			for (Catalog catalog : mojo.readCatalog(helmCatalogFile)) {
442 				assertEquals(archive.toString(), catalog.getChart().toString());
443 				assertEquals(uploadUrl, catalog.getUploadUrl().toString());
444 				assertNull(catalog.getUploadResponseType());
445 				assertEquals("", catalog.getUploadResponse());
446 			}
447 		} catch (MojoExecutionException e) {
448 			fail(e);
449 		}
450

465 	private void mockHelmShowChart() {
466 		mock.stubFor(WireMock.put(WireMock.urlMatching(".*"))
467 				.willReturn(WireMock.ok()
468 						.withStatus(200)));
469 		mock.stubFor(WireMock.get(WireMock.urlMatching(".*/index\\.yaml$"))
470 				.willReturn(WireMock.ok()
471 						.withStatus(200)
