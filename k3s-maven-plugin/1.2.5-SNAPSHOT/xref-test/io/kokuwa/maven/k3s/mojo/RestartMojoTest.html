3   import static org.junit.jupiter.api.Assertions.assertDoesNotThrow;
4   import static org.junit.jupiter.api.Assertions.assertThrowsExactly;
5   
6   import java.util.List;
7   
8   import org.apache.maven.plugin.MojoExecutionException;
9   import org.junit.jupiter.api.DisplayName;
10  import org.junit.jupiter.api.Test;
11  
12  import io.kokuwa.maven.k3s.test.AbstractTest;
13  
14  /**
15   * Test for {@link RestartMojo}.
16   *
17   * @author stephan.schnabel@posteo.de
18   */
19  @DisplayName("mojo: restart")
20  public class RestartMojoTest extends AbstractTest {
21  
22  	@DisplayName("with skip")
23  	@Test
24  	void withSkip(RestartMojo restartMojo) {
25  
26  		restartMojo.setSkipRestart(false);
27  		restartMojo.setSkip(true);
28  		assertDoesNotThrow(restartMojo::execute);
29  
30  		restartMojo.setSkipRestart(true);
31  		restartMojo.setSkip(false);
32  		assertDoesNotThrow(restartMojo::execute);
33  
34  		restartMojo.setSkipRestart(true);
35  		restartMojo.setSkip(true);
36  		assertDoesNotThrow(restartMojo::execute);
37  	}
38  
39  	@DisplayName("without container")
40  	@Test
41  	void withoutContainer(RestartMojo restartMojo) {
42  		restartMojo.setResources(List.of("pod/nope"));
43  		assertThrowsExactly(MojoExecutionException.class, restartMojo::execute, () -> "No k3s container found");
44  	}
45  
46  	@DisplayName("without resources")
47  	@Test
48  	void withoutResources(RestartMojo restartMojo) {
49  		assertDoesNotThrow(restartMojo::execute);
50  	}
51  
52  	@DisplayName("without invalid resource")
53  	@Test
54  	void invalid(RunMojo runMojo, RestartMojo restartMojo) {
55  		assertDoesNotThrow(runMojo::execute);
56  		restartMojo.setResources(List.of("pod/nope"));
57  		assertThrowsExactly(MojoExecutionException.class, restartMojo::execute, () -> "No k3s container found");
58  		restartMojo.setResources(List.of("deployment/nope/nope"));
59  		assertThrowsExactly(MojoExecutionException.class, restartMojo::execute, () -> "No k3s container found");
60  	}
61  
62  	@DisplayName("with statefulset")
63  	@Test
64  	void statefulset(RunMojo runMojo, ApplyMojo applyMojo, RestartMojo restartMojo) {
65  		applyMojo.setSubdir("statefulset");
66  		restartMojo.setResources(List.of("statefulset/echo"));
67  		assertDoesNotThrow(runMojo::execute);
68  		assertDoesNotThrow(applyMojo::execute);
69  		assertDoesNotThrow(restartMojo::execute);
70  	}
71  
72  	@DisplayName("with not existing resource")
73  	@Test
74  	void notExisting(RunMojo runMojo, RestartMojo restartMojo) {
75  		restartMojo.setResources(List.of("statefulset/echo"));
76  		assertDoesNotThrow(runMojo::execute);
77  		assertThrowsExactly(MojoExecutionException.class, restartMojo::execute, () -> "Failed to restart resources");
78  	}
79  }
