123 	@Override
124 	public void execute() throws MojoExecutionException {
125 
126 		if (skip || skipUpload) {
127 			getLog().info("Skip upload");
128 			return;
129 		}
130 
131 		if (uploadVerificationTimeout != null && uploadVerificationTimeout <= 0) {
132 			throw new MojoExecutionException("Timeout must be a positive value.");
133 		}
134 
135 		getLog().info("Uploading to " + getHelmUploadUrl() + "\n");
136 		for (Path chart : getChartArchives()) {
137 			getLog().info("Uploading " + chart + "...");
138 			try {
139 				uploadSingle(chart);
140 			} catch (IOException e) {
141 				throw new MojoExecutionException("Upload failed.", e);
142 			}
143 		}
144 
145 		if (!skipCatalog) {
146 			Path catalogPath = getCatalogFilePath();
147 			getLog().info("Attaching catalog artifact: " + catalogPath);
148 			mavenProjectHelper.attachArtifact(mavenProject, CATALOG_ARTIFACT_TYPE, CATALOG_ARTIFACT_NAME,
149 					catalogPath.toFile());
150 		}
151 
152 		if (uploadVerification) {
153 			for (Path chartDirectory : getChartDirectories()) {
154 				Path chartPath = chartDirectory.resolve("Chart.yaml");
155 				getLog().info("Verifying upload of " + chartPath);
156 				if (!verifyUpload(chartPath)) {
157 					getLog().info("Upload verification timed out.");
158 					throw new MojoExecutionException("Chart verification failed");
159 				}
160 			}
161 		}
162 	}

164 	/**
165 	 * Returns the proper upload URL based on the provided chart version. Charts w/ an SNAPSHOT suffix will be uploaded
166 	 * to SNAPSHOT repo.
167 	 *
168 	 * @return Upload URL based on chart version
169 	 */
170 	private String getHelmUploadUrl() {
171 		String chartVersion = getChartVersion();
172 		HelmRepository uploadRepoStable = getUploadRepoStable();
173 		HelmRepository uploadRepoSnapshot = getUploadRepoSnapshot();
174 
175 		String uploadUrl = uploadRepoStable.getUrl();
176 		if (chartVersion != null && chartVersion.endsWith("-SNAPSHOT")
177 				&& uploadRepoSnapshot != null
178 				&& StringUtils.isNotEmpty(uploadRepoSnapshot.getUrl())) {
179 			uploadUrl = uploadRepoSnapshot.getUrl();
180 		}
181 
182 		return uploadUrl;
183 	}

185 	/**
186 	 * Reads the catalog file and deserializes its content as a POJO.
187 	 *
188 	 * @param catalogFile instance of catalog file
189 	 * @return list of Catalog
190 	 * @throws MojoExecutionException when IOException is encountered
191 	 */
192 	List<Catalog> readCatalog(File catalogFile) throws MojoExecutionException {
193 		List<Catalog> catalogList = new ArrayList<>();
194 		if (catalogFile == null || !catalogFile.exists()) {
195 			return catalogList;
196 		}
197 		try {
198 			return Arrays.asList(MAPPER.readValue(catalogFile, Catalog[].class));
199 		} catch (DatabindException e) {
200 			getLog().warn("Unable to parse the existing catalog file content. Overwriting data.");
201 		} catch (IOException e) {
202 			throw new MojoExecutionException("Failure occurred while reading the catalog file.", e);
203 		}
204 		return catalogList;
205 	}

207 	/**
208 	 * Reads the existing helm catalog content file and merges the new catalog data with it.
209 	 *
210 	 * @param data helm chart upload info represented as the Catalog object
211 	 * @return pretty string json representation of the updated helm catalog contents
212 	 */
213 	private String createCatalogContent(Catalog data) throws MojoExecutionException {
214 		File file = getCatalogFilePath().toFile();
215 		List<Catalog> catalog = readCatalog(file);
216 		catalog.add(data);
217 		try {
218 			return MAPPER.writeValueAsString(catalog);
219 		} catch (JsonProcessingException e) {
220 			throw new MojoExecutionException("Failure occurred while writing the catalog content as a JSON string.", e);
221 		}
222 	}

236 	 * Writes the catalog data to the catalog file.
237 	 *
238 	 * @param content Helm upload catalog data in string form
239 	 * @throws MojoExecutionException when writing catalog data to the file
240 	 */
241 	private void catalogHelmChart(String content) throws MojoExecutionException {
242 		Path catalogPath = getCatalogFilePath();
243 		File catalogFile = catalogPath.toFile();
244 		getLog().debug("Writing content to catalog file: " + content);
245 
246 		try (OutputStream out = Files.newOutputStream(catalogFile.toPath())) {
247 			// FIXME: It is better to use the encoding either from
248 			// project.build.sourceEncoding or project.reporting.outputEncoding
249 			OutputStreamWriter fileWriter = new OutputStreamWriter(out, StandardCharsets.UTF_8);
250 			fileWriter.write(content);
251 			fileWriter.close();
252 		} catch (IOException e) {
253 			throw new MojoExecutionException("Failure occurred while writing the catalog file.", e);
254 		}
255 	}

257 	private void uploadSingle(Path chart) throws MojoExecutionException, IOException {
258 		File fileToUpload = chart.toFile();
259 		HelmRepository uploadRepo = getHelmUploadRepo();
260 
261 		HttpURLConnection connection;
262 		if (uploadRepo.getType() == null) {
263 			throw new IllegalArgumentException("Repository type missing. Check your plugin configuration.");
264 		}
265 
266 		switch (uploadRepo.getType()) {
267 			case ARTIFACTORY:
268 				connection = getConnectionForUploadToArtifactory(fileToUpload, uploadRepo);
269 				break;
270 			case CHARTMUSEUM:
271 				connection = getConnectionForUploadToChartMuseum();
272 				break;
273 			case NEXUS:
274 				connection = getConnectionForUploadToNexus(fileToUpload);
275 				break;
276 			default:
277 				throw new IllegalArgumentException("Unsupported repository type for upload.");
278 		}
279 
280 		if (insecure && connection instanceof HttpsURLConnection) {
281 			getLog().info("Use insecure TLS connection for [" + connection.getURL() + "]");
282 			TLSHelper.insecure((HttpsURLConnection) connection);
283 		}
284 
285 		try (FileInputStream fileInputStream = new FileInputStream(fileToUpload)) {
286 			IOUtils.copy(fileInputStream, connection.getOutputStream());
287 		}
288 		if (connection.getResponseCode() >= 300) {
289 			String response;
290 			if (connection.getErrorStream() != null) {
291 				response = new String(IOUtils.toByteArray(connection.getErrorStream()), StandardCharsets.UTF_8);
292 			} else if (connection.getInputStream() != null) {
293 				response = new String(IOUtils.toByteArray(connection.getInputStream()), StandardCharsets.UTF_8);
294 			} else {
295 				response = "No details provided";
296 			}
297 			throw new MojoExecutionException("Failed to upload: " + response);
298 		}
299 		String message = "Returned: " + connection.getResponseCode();
300 		String details = "";
301 		if (connection.getInputStream() != null) {
302 			details = new String(IOUtils.toByteArray(connection.getInputStream()), StandardCharsets.UTF_8);
303 			if (!details.isEmpty()) {
304 				message += " - " + details;
305 			}
306 		}
307 		getLog().info(message);
308 		if (!skipCatalog) {
309 			Catalog data = new Catalog(chart, connection.getURL(), connection.getContentType(), details);
310 			catalogHelmChart(createCatalogContent(data));
311 		}
312 		connection.disconnect();
313 	}

315 	private boolean verifyUpload(Path chartPath) throws MojoExecutionException {
316 		String chartName;
317 		try {
318 			chartName = YAML_MAPPER.readValue(chartPath.toFile(), HelmChart.class).getName();
319 		} catch (IOException e) {
320 			throw new MojoExecutionException("Unable to read chart from " + chartPath, e);
321 		}
322 
323 		long startTimeMillis = System.currentTimeMillis();
324 		long timeoutMillis = uploadVerificationTimeout * 1000;
325 		long cutoffMillis = startTimeMillis + timeoutMillis;
326 		boolean verificationSuccess = false;
327 
328 		while (System.currentTimeMillis() < cutoffMillis && !verificationSuccess) {
329 			try {
330 				helm()
331 					.arguments("show", "chart", chartName,
332 						"--version", getChartVersion(), "--repo", getHelmUploadUrl())
333 					.execute("show chart failed");
334 				verificationSuccess = true;
335 			} catch (Exception MojoExecutionException) {
336 				getLog().info("Upload verification failed, retrying...");
337 				try {
338 					Thread.sleep(1000);
339 				} catch (InterruptedException ie) {
340 					throw new MojoExecutionException("Upload verification interrupted", ie);
341 				}
342 			}
343 		}
344 		return verificationSuccess;
345 	}

347 	private HttpURLConnection getConnectionForUploadToChartMuseum() throws IOException, MojoExecutionException {
348 		HttpURLConnection connection = (HttpURLConnection) new URL(getHelmUploadUrl()).openConnection();
349 		connection.setDoOutput(true);
350 		connection.setRequestMethod("POST");
351 		connection.setRequestProperty("Content-Type", "application/gzip");
352 
353 		setBasicAuthHeader(connection);
354 
355 		return connection;
356 	}

358 	private void setBasicAuthHeader(HttpURLConnection connection) throws MojoExecutionException {
359 		PasswordAuthentication authentication = getAuthentication(getHelmUploadRepo());
360 		if (authentication != null) {
361 			String encoded = Base64.getEncoder()
362 					.encodeToString((authentication.getUserName() + ":" + new String(authentication.getPassword()))
363 							.getBytes(StandardCharsets.UTF_8));
364 			connection.setRequestProperty("Authorization", "Basic " + encoded);
365 		}
366 	}

368 	private HttpURLConnection getConnectionForUploadToArtifactory(File file, HelmRepository repo)
369 			throws IOException, MojoExecutionException {
370 		String uploadUrl = getHelmUploadUrl();
371 		// Append slash if not already in place
372 		if (!uploadUrl.endsWith("/")) {
373 			uploadUrl += "/";
374 		}
375 		if (repo.isUseGroupId()) {
376 			uploadUrl += projectGroupId.replace(".", "/") + "/";
377 		}
378 		if (repo.isUseArtifactId()) {
379 			uploadUrl += projectArtifactId.replace(".", "/") + "/";
380 		}
381 		if (repo.isUseGroupId() || repo.isUseArtifactId()) {
382 			uploadUrl += projectVersion + "/";
383 		}
384 
385 		uploadUrl = uploadUrl + file.getName();
386 
387 		HttpURLConnection connection = (HttpURLConnection) new URL(uploadUrl).openConnection();
388 		connection.setDoOutput(true);
389 		connection.setRequestMethod("PUT");
390 		connection.setRequestProperty("Content-Type", "application/gzip");
391 
392 		verifyAndSetAuthentication(true);
393 
394 		return connection;
395 	}

415 	/**
416 	 *
417 	 * @param requireCredentials The need for credentials depends on how the repository is configured. For instance on
418 	 *                           nexus it is possible to configure a repository without authentication
419 	 * @throws MojoExecutionException
420 	 */
421 	private void verifyAndSetAuthentication(boolean requireCredentials) throws MojoExecutionException {
422 
423 		PasswordAuthentication authentication = getAuthentication(getHelmUploadRepo());
424 		if (authentication != null) {
425 			Authenticator.setDefault(new Authenticator() {
426 				@Override
427 				protected PasswordAuthentication getPasswordAuthentication() {
428 					return authentication;
429 				}
430 			});
431 		} else if (requireCredentials) {
432 			throw new IllegalArgumentException("Credentials has to be configured for uploading to Artifactory.");
433 		}
434 	}
